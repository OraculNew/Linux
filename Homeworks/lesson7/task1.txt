Подключить репозиторий с nginx любым удобным способом, установить nginx и потом удалить nginx, используя утилиту dpkg.

oracul@oracul-VirtualBox:~$ cd /etc/apt/
oracul@oracul-VirtualBox:/etc/apt$ nano sources.list
oracul@oracul-VirtualBox:/etc/apt$ sudo nano sources.list

#deb cdrom:[Ubuntu 20.04.1 LTS _Focal Fossa_ - Release amd64 (20200731)]/ focal main restricted
deb http://nginx.org/packages/ubuntu focal nginx
-//-

oracul@oracul-VirtualBox:/etc/apt$ sudo apt install curl

oracul@oracul-VirtualBox:/etc/apt$ curl -fsSL https://nginx.org/keys/nginx_signing.key | sudo apt-key add -
OK
