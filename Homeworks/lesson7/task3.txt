Настроить iptables: разрешить подключения только на 22-й и 80-й порты.

oracul@oracul-VirtualBox:/$ sudo iptables -P INPUT DROP
oracul@oracul-VirtualBox:/$ sudo iptables -P OUTPUT DROP
oracul@oracul-VirtualBox:/$ sudo iptables -P FORWARD DROP
oracul@oracul-VirtualBox:/$ sudo iptables -A INPUT -i lo -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A OUTPUT -o lo -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A INPUT -p icmp -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A OUTPUT -p icmp -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A OUTPUT -p TCP -m tcp --sport 32768:61000 -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A OUTPUT -p UDP -m udp --sport 32768:61000 -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A INPUT -p TCP -m state --state ESTABLISHED,RELATED -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A INPUT -p UDP -m state --state ESTABLISHED,RELATED -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A INPUT -p tcp -m tcp --dport 22 -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A OUTPUT -p tcp -m tcp --sport 22 -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A INPUT -p tcp -m tcp --dport 80 -j ACCEPT
oracul@oracul-VirtualBox:/$ sudo iptables -A OUTPUT -p tcp -m tcp --sport 80 -j ACCEPT
