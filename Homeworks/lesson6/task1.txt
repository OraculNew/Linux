1. Написать скрипт, который удаляет из текстового файла пустые строки и заменяет маленькие символы на большие.
Воспользуйтесь tr или SED.

oracul@oracul-VirtualBox:~$ cat > task1.sh
#!/bin/bash
tr -s '\n' < $1 | tr [:lower:] [:upper:] > $2

oracul@oracul-VirtualBox:~$ chmod +x task1.sh

oracul@oracul-VirtualBox:~$ cat > input.txt
ewrwqr rqwrqw rwqesdg aKLL  JGh  Hhgjsdf

oracul@oracul-VirtualBox:~$ ./task1.sh input.txt output.txt
oracul@oracul-VirtualBox:~$ cat output.txt
EWRWQR RQWRQW RWQESDG AKLL  JGH  HHGJSDF
